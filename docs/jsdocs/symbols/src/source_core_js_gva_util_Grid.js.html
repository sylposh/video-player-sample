 17 /**
 18  * Provide a namespace for this class
 19  */
 20 goog.provide('gva.util.Grid');
 21 
 22 /**
 23  * Creates a new Grid instance.
 24  * @constructor
 25  */
 26 gva.util.Grid = function() {
 27 
 28     /**
 29      * The number of rows along the y axis of the grid.
 30      * @type {number}
 31      */
 32     this.rows = 0;
 33 
 34     /**
 35      * The number of columns along the x axis of the grid.
 36      * @type {number}
 37      */
 38     this.cols = 0;
 39 
 40     /**
 41      * The total number of cells available in the grid
 42      * @type {number}
 43      */
 44     this.cells = 0;
 45 
 46     /**
 47      * An array of 2D points representing the top left position of each grid cell.
 48      * @type {Array.<Object>}
 49      */
 50     this.coords = [];
 51 
 52     /**
 53      * Whether columns should be positioned in the center of the available width
 54      * @type {Boolean}
 55      */
 56     this.centerAlignX = true;
 57 
 58     /**
 59      * Whether rows should be positioned in the center of the available height
 60      * @type {Boolean}
 61      */
 62     this.centerAlignY = true;
 63 
 64     /**
 65      * The outer boundaries of the grid
 66      * @type {Object}
 67      */
 68     this.gridBounds = {
 69 
 70             x: 1.0,
 71             y: 1.0,
 72             width: 1.0,
 73             height: 1.0
 74     };
 75 
 76     /**
 77      * The outer boundaries of each grid cell
 78      * @type {Object}
 79      */
 80     this.cellBounds = {
 81 
 82             width: 1.0,
 83             height: 1.0,
 84             spacingX: 1.0,
 85             spacingY: 1.0
 86     };
 87 };
 88 goog.inherits(gva.util.Grid, goog.events.EventTarget);
 89 
 90 /**
 91  * Sets the outer boundaries of the grid and automatically recalculates the cell positions.
 92  * @param {number} x The position of the grid on the x axis.
 93  * @param {number} y The position of the grid on the y axis.
 94  * @param {number} width The total width of the grid.
 95  * @param {number} height The total height of the grid.
 96  */
 97 gva.util.Grid.prototype.setGridBounds = function(x, y, width, height) {
 98     this.gridBounds.x = isNaN(x) ? 1.0 : parseFloat(x);
 99     this.gridBounds.y = isNaN(y) ? 1.0 : parseFloat(y);
100     this.gridBounds.width = isNaN(width) ? 1.0 : parseFloat(width);
101     this.gridBounds.height = isNaN(height) ? 1.0 : parseFloat(height);
102     this.refresh();
103 };
104 
105 /**
106  * Sets the outer boundaries of each grid cell and automatically recalculates the cell positions.
107  * @param {number} width The width of each grid cell.
108  * @param {number} height The height of each grid cell.
109  * @param {number} spacingX The spacing between grid cells along the x axis.
110  * @param {number} spacingY The spacing between grid cells along the y axis.
111  */
112 gva.util.Grid.prototype.setCellBounds = function(width, height, spacingX, spacingY) {
113     this.cellBounds.width = isNaN(width) ? 1.0 : parseFloat(width);
114     this.cellBounds.height = isNaN(height) ? 1.0 : parseFloat(height);
115     this.cellBounds.spacingX = isNaN(spacingX) ? 1.0 : parseFloat(spacingX);
116     this.cellBounds.spacingY = isNaN(spacingY) ? 1.0 : parseFloat(spacingY);
117     this.refresh();
118 };
119 
120 /**
121  * Recalculates each cell position based on the current grid setup
122  */
123 gva.util.Grid.prototype.refresh = function() {
124 
125     // Store the total with and height of each cell, including spacing
126     var stepX = this.cellBounds.width + this.cellBounds.spacingX;
127     var stepY = this.cellBounds.height + this.cellBounds.spacingY;
128 
129     // Compute the maximum possible number of cells in each direction
130     this.cols = Math.floor((this.gridBounds.width - this.cellBounds.spacingX) / stepX);
131     this.rows = Math.floor((this.gridBounds.height - this.cellBounds.spacingY) / stepY);
132 
133     // Compute the remaining space after accommodating the cells
134     var spaceX = (this.gridBounds.width - this.cellBounds.spacingX) - (this.cols * stepX);
135     var spaceY = (this.gridBounds.height - this.cellBounds.spacingY) - (this.rows * stepY);
136 
137     // Offset by half the remaining space
138     var startX = this.centerAlignX ? this.cellBounds.spacingX + spaceX * 0.5 : 0.0;
139     var startY = this.centerAlignY ? this.cellBounds.spacingY + spaceY * 0.5 : 0.0;
140 
141     // Calculate coordinates from dimensions
142     this.coords = [];
143 
144     var row, col;
145 
146     for (row = 0; row < this.rows; row++) {
147         for (col = 0; col < this.cols; col++) {
148             this.coords.push({
149                 x: startX + (col * stepX),
150                 y: startY + (row * stepY)
151             });
152         }
153     }
154 
155     // Update the total amount of visible cells
156     this.cells = this.rows * this.cols;
157 };
